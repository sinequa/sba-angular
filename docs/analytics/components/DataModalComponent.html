import { Component, Inject } from "@angular/core";
import { Action } from "@sinequa/components/action";
import { UIService } from "@sinequa/components/utils";
import { FormatService, ValueItem } from "@sinequa/core/app-utils";
import { FieldValue } from "@sinequa/core/base";
import { MODAL_MODEL } from "@sinequa/core/modal";
import { CCColumn } from "@sinequa/core/web-services";

declare type DataModel = {
    cell: ValueItem | FieldValue,
    row: any,
    column: string,
    cccolumn: CCColumn,
    formatContent: boolean
};

@Component({
    selector: 'sq-data-modal',
    templateUrl: './data-modal.component.html',
    styles: [`
pre {
    overflow: auto;
    font-size: 12px;
    max-height: 50vh;
}
    `]
})
export class DataModalComponent {

    copyAction: Action;
    formatContent = true;
    data: string;

    constructor(
        @Inject(MODAL_MODEL) public model: DataModel,
        public uiService: UIService,
        public formatService: FormatService
    ){
        this.formatContent = model.formatContent;
    }

    ngOnInit() {

        this.copyAction = new Action({
            icon: "far fa-copy",
            title: "Copy to Clipboard",
            action: () => this.uiService.copyToClipboard(this.data)
        });

        this.updateData();
    }

    updateData() {
        if(this.formatContent) {
            this.data = this.formatService.formatValue(this.model.cell, this.model.cccolumn);
        }
        else {
            this.data = this.formatService.formatRaw(this.model.cell);
        }
    }
}
<sq-modal [title]="model.column">
    <div class="custom-control custom-checkbox mb-2">
        <input type="checkbox" class="custom-control-input" id="format-popup" [(ngModel)]="formatContent" (ngModelChange)="updateData()">
        <label class="custom-control-label" for="format-popup">{{'msg#grid.formatData' | sqMessage}}</label>
    </div>

    <pre class="border mb-0 p-1">{{data}}</pre>

    <div footer class="btn-group"
        [sq-action-buttons]="{
            items: [copyAction],
            rightAligned: true,
            style: 'primary'
        }">
    </div>
</sq-modal>
